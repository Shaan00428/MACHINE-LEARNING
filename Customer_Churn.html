In[23]:
def bulid_model(input_shape):
    model = Sequential([
        Input (shape=input_shape),
        Dense(64,activation='relu'), #1st hidden layer
        Dropout(0.5),
        Dense(32,activation= 'relu'), #2nd hidden layer
        Dropout(0.5),
        Dense(1,activation='sigmoid') #output layer
    ])
    return model

input_shape = (x_train.shape[1],)
model = bulid_model(input_shape)

#plot training history 
def plot_history(histroy,metric):
    plt.figure(figsize=(10,6))
    plt.plot(histroy.history[metric],label=f'Training{metric}')
    plt.plot(histroy.history[f'val_{metric}'],label=f'Validation{metric}')
    plt.xlabel('Epochs')
    plt.ylabel(metric.capitalize())
    plt.title (f'Training and Validation{metric.capitalize()}')
    plt.legend()
    plt.show()

plot_history (history,'accuracy')
plot_history (history,'loss')
